030package org.openimaj.demos.image;
031
032import java.io.IOException;
033
034import org.openimaj.demos.Demo;
035import org.openimaj.image.DisplayUtilities;
036import org.openimaj.image.FImage;
037import org.openimaj.image.ImageUtilities;
038import org.openimaj.image.MBFImage;
039
040/**
041 *      Demonstration of the MBFImage. Displays an image along with its red, 
042 *      green and blue channels. The code shows how these images can be written 
043 *      to files.
044 *              
045 *  @author David Dupplaw (dpd@ecs.soton.ac.uk)
046 *      
047 *      @created 15 Feb 2012
048 */
049@Demo(
050        author = "", 
051        description = "Demonstration of the MBFImage. Displays an image along " +
052                        "with its red, green and blue channels. The code shows how these " +
053                        "images can be written to files.", 
054        keywords = { "image", "colour", "bands" }, 
055        title = "Colour Image Demo",
056        icon = "/org/openimaj/demos/icons/image/bird-icon.png"
057)
058public class ImageRepresentation {
059        /**
060         *      Default main
061         *  @param args Command-line arguments
062         *  @throws IOException
063         */
064        public static void main(String[] args) throws IOException {
065                MBFImage image = ImageUtilities.readMBF(
066                                ImageRepresentation.class.getResourceAsStream(
067                                                "/org/openimaj/demos/image/bird.png"));
068                DisplayUtilities.display(image);
069                
070                FImage blank = new FImage(image.getWidth(), image.getHeight());
071                MBFImage red = new MBFImage(image.getBand(0), blank, blank);
072                MBFImage green = new MBFImage(blank, image.getBand(1), blank);
073                MBFImage blue = new MBFImage(blank, blank, image.getBand(2));
074                
075                DisplayUtilities.displayLinked( "Images", 2, image, red, green, blue );         
076        }
077}
