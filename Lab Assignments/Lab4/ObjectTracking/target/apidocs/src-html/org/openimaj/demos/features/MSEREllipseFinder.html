030package org.openimaj.demos.features;
031
032import java.util.ArrayList;
033import java.util.List;
034
035import org.openimaj.demos.Demo;
036import org.openimaj.image.DisplayUtilities;
037import org.openimaj.image.MBFImage;
038import org.openimaj.image.analysis.watershed.Component;
039import org.openimaj.image.analysis.watershed.feature.MomentFeature;
040import org.openimaj.image.colour.ColourSpace;
041import org.openimaj.image.colour.RGBColour;
042import org.openimaj.image.colour.Transforms;
043import org.openimaj.image.feature.local.detector.mser.MSERFeatureGenerator;
044import org.openimaj.image.renderer.MBFImageRenderer;
045import org.openimaj.math.geometry.shape.Ellipse;
046
047/**
048 * Demo showing ellipse fitted to MSERs
049 *
050 * @author Sina Samangooei (ss@ecs.soton.ac.uk)
051 */
052@Demo(
053                author = "Sina Samangooei",
054                description = "Shows ellipse fitting for regions detected using an MSER feature extractor",
055                keywords = {
056                                "mser", "ellipse", "feat" },
057                title = "MSER Ellipse Finder")
058public class MSEREllipseFinder {
059        /**
060         * Construct demo
061         */
062        public MSEREllipseFinder() {
063                final MBFImage image = new MBFImage(400, 400, ColourSpace.RGB);
064                final MBFImageRenderer renderer = image.createRenderer();
065
066                image.fill(RGBColour.WHITE);
067                final List<Ellipse> ellipses = new ArrayList<Ellipse>();
068                ellipses.add(new Ellipse(200, 100, 100, 80, Math.PI / 4));
069                ellipses.add(new Ellipse(200, 300, 50, 30, -Math.PI / 4));
070                ellipses.add(new Ellipse(100, 300, 30, 50, -Math.PI / 3));
071
072                for (final Ellipse ellipse : ellipses) {
073                        renderer.drawShapeFilled(ellipse, RGBColour.BLACK);
074                }
075
076                final MSERFeatureGenerator mser = new MSERFeatureGenerator(MomentFeature.class);
077                final List<Component> features = mser.generateMSERs(Transforms
078                                .calculateIntensityNTSC(image));
079                for (final Component c : features) {
080                        final MomentFeature feature = c.getFeature(MomentFeature.class);
081                        renderer.drawShape(feature.getEllipse(2), RGBColour.RED);
082                        renderer.drawShape(feature.getEllipse(2)
083                                        .calculateOrientedBoundingBox(), RGBColour.GREEN);
084                }
085                DisplayUtilities.display(image);
086        }
087
088        /**
089         * The main method
090         *
091         * @param args
092         *            ignored
093         */
094        public static void main(String args[]) {
095                new MSEREllipseFinder();
096        }
097}
